<script>
    // State
    const state = {
        sources: [],
        customSources: '',
        topics: [],
        institutions: '',
        voices: '',
        time: '06:00',
        email: ''
    };

    // DOM Elements
    const steps = {
        step1: document.getElementById('step1'),
        step2: document.getElementById('step2'),
        step3: document.getElementById('step3'),
        success: document.getElementById('successState')
    };

    const progressFill = document.getElementById('progressFill');

    // Chip selection
    document.querySelectorAll('.chip').forEach(chip => {
        chip.addEventListener('click', () => {
            chip.classList.toggle('selected');
            updateState();
            updateButtons();
        });
    });

    // Time selection
    document.querySelectorAll('.time-option').forEach(option => {
        option.addEventListener('click', () => {
            document.querySelectorAll('.time-option').forEach(o => o.classList.remove('selected'));
            option.classList.add('selected');
            state.time = option.dataset.time;
        });
    });

    // Update state from chips
    function updateState() {
        state.sources = [];
        state.topics = [];

        document.querySelectorAll('[data-group="indonesia"] .chip.selected, [data-group="asia"] .chip.selected, [data-group="global"] .chip.selected').forEach(chip => {
            state.sources.push(chip.dataset.value);
        });

        document.querySelectorAll('[data-group="topics"] .chip.selected').forEach(chip => {
            state.topics.push(chip.dataset.value);
        });
    }

    // Update button states
    function updateButtons() {
        document.getElementById('next1').disabled = state.sources.length === 0;
        document.getElementById('next2').disabled = state.topics.length === 0;
    }

    // Navigation
    document.getElementById('next1').addEventListener('click', () => {
        state.customSources = document.getElementById('customSourcesInput').value;
        goToStep(2);
    });

    document.getElementById('back2').addEventListener('click', () => {
        goToStep(1);
    });

    document.getElementById('next2').addEventListener('click', () => {
        state.institutions = document.getElementById('institutionsInput').value;
        state.voices = document.getElementById('voicesInput').value;
        updateSummary();
        goToStep(3);
    });

    document.getElementById('back3').addEventListener('click', () => {
        goToStep(2);
    });

    document.getElementById('submitBtn').addEventListener('click', () => {
        const email = document.getElementById('emailInput').value;
        if (!email || !email.includes('@')) {
            alert('Mohon masukkan email yang valid');
            return;
        }
        state.email = email;
        
        // Show success
        document.getElementById('successTime').textContent = state.time;
        document.getElementById('successEmail').textContent = state.email;
        
        steps.step3.classList.remove('active');
        steps.success.classList.add('active');
        progressFill.style.width = '100%';

        // Here you would send data to backend
        console.log('Submitted:', state);
    });

    function goToStep(stepNum) {
        Object.values(steps).forEach(s => s.classList.remove('active'));
        steps[`step${stepNum}`].classList.add('active');
        progressFill.style.width = `${(stepNum / 3) * 100}%`;
    }

    function updateSummary() {
        // Sources summary (including custom)
        let allSources = [...state.sources];
        if (state.customSources.trim()) {
            allSources.push(state.customSources.trim());
        }
        document.getElementById('summarySources').textContent = 
            allSources.length > 3 
                ? allSources.slice(0, 3).join(', ') + ` +${allSources.length - 3} lainnya`
                : allSources.join(', ');
        
        // Topics summary
        document.getElementById('summaryTopics').textContent = 
            state.topics.length > 3 
                ? state.topics.slice(0, 3).join(', ') + ` +${state.topics.length - 3} lainnya`
                : state.topics.join(', ');

        // Voices summary (only show if filled)
        const voicesRow = document.getElementById('summaryVoicesRow');
        if (state.voices.trim() || state.institutions.trim()) {
            voicesRow.style.display = 'block';
            const combined = [state.institutions, state.voices].filter(v => v.trim()).join(', ');
            document.getElementById('summaryVoices').textContent = combined;
        } else {
            voicesRow.style.display = 'none';
        }
    }
</script>
