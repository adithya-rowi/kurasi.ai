## Update 3: Dashboard Greeting (Indonesian)

Update the dashboard to use Indonesian greetings based on time:

```typescript
function getIndonesianGreeting(name: string): string {
  const hour = new Date().getHours();
  
  if (hour >= 5 && hour < 11) {
    return `Selamat pagi, ${name}!`;
  } else if (hour >= 11 && hour < 15) {
    return `Selamat siang, ${name}!`;
  } else if (hour >= 15 && hour < 18) {
    return `Selamat sore, ${name}!`;
  } else {
    return `Selamat malam, ${name}!`;
  }
}
```
