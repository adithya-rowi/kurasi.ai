### 1. Create Dashboard Components

Create file: `client/src/components/Dashboard/BriefCard.tsx`

```tsx
import { useState } from 'react';
import { ChevronDown, ChevronUp, ExternalLink, Bookmark, ThumbsDown } from 'lucide-react';

interface BriefItem {
  title: string;
  summary: string;
  source: string;
  url: string;
  whyItMatters: string;
  foundByModels?: string[];
  verificationScore?: number;
}

interface BriefCardProps {
  item: BriefItem;
  priority: 'critical' | 'important' | 'background';
  onSave?: (item: BriefItem) => void;
  onNotRelevant?: (item: BriefItem) => void;
}

export function BriefCard({ item, priority, onSave, onNotRelevant }: BriefCardProps) {
  const [expanded, setExpanded] = useState(priority === 'critical');
  
  const priorityStyles = {
    critical: {
      border: 'border-l-4 border-l-red-500',
      bg: 'bg-red-50',
      badge: 'bg-red-100 text-red-700'
    },
    important: {
      border: 'border-l-4 border-l-yellow-500',
      bg: 'bg-yellow-50',
      badge: 'bg-yellow-100 text-yellow-700'
    },
    background: {
      border: 'border-l-4 border-l-green-500',
      bg: 'bg-green-50',
      badge: 'bg-green-100 text-green-700'
    }
  };
  
  const styles = priorityStyles[priority];
  
  return (
    <div className={`${styles.border} ${styles.bg} rounded-lg mb-4 overflow-hidden transition-all duration-200`}>
      {/* Header - Always visible */}
      <div 
        className="p-4 cursor-pointer hover:bg-opacity-80"
        onClick={() => setExpanded(!expanded)}
      >
        <div className="flex items-start justify-between">
          <div className="flex-1">
            <h3 className="font-semibold text-gray-900 text-lg leading-tight">
              {item.title}
            </h3>
            
            {!expanded && (
              <p className="text-gray-600 text-sm mt-1 line-clamp-2">
                {item.summary}
              </p>
            )}
          </div>
          
          <button className="ml-4 text-gray-400 hover:text-gray-600">
            {expanded ? <ChevronUp size={20} /> : <ChevronDown size={20} />}
          </button>
        </div>
        
        {/* Source badges - Always visible */}
        <div className="flex items-center gap-2 mt-2">
          <span className="text-xs text-gray-500 bg-white px-2 py-1 rounded">
            üì∞ {item.source}
          </span>
          
          {item.foundByModels && item.foundByModels.length > 1 && (
            <span className={`text-xs px-2 py-1 rounded ${styles.badge}`}>
              ‚úì {item.foundByModels.length} models agree
            </span>
          )}
          
          {item.verificationScore && item.verificationScore >= 8 && (
            <span className="text-xs text-green-700 bg-green-100 px-2 py-1 rounded">
              High confidence
            </span>
          )}
        </div>
      </div>
      
      {/* Expanded content */}
      {expanded && (
        <div className="px-4 pb-4 border-t border-gray-200 bg-white">
          {/* Full summary */}
          <p className="text-gray-700 mt-4 leading-relaxed">
            {item.summary}
          </p>
          
          {/* Why it matters - THE KEY PERSONALIZATION */}
          <div className="mt-4 p-3 bg-blue-50 rounded-lg border border-blue-100">
            <p className="text-sm font-medium text-blue-800 mb-1">
              üí° Why this matters to you:
            </p>
            <p className="text-sm text-blue-700">
              {item.whyItMatters}
            </p>
          </div>
          
          {/* Actions */}
          <div className="flex items-center justify-between mt-4 pt-4 border-t">
            <div className="flex items-center gap-3">
              {item.url && item.url !== 'search required' && (
                <a
                  href={item.url}
                  target="_blank"
                  rel="noopener noreferrer"
                  className="flex items-center gap-1 text-blue-600 hover:text-blue-800 text-sm font-medium"
                  onClick={(e) => e.stopPropagation()}
                >
```

### 3. Track User Interactions (TikTok Algorithm)

Create file: `server/routes/interactions.ts`

```typescript
import { Router } from 'express';
import { db } from '../db';

const router = Router();

// Track any user interaction
router.post('/api/interactions', async (req, res) => {
  try {
    const userId = req.user?.id;
    const { type, articleTitle, articleId, source, timeSpent, scrollDepth } = req.body;
    
    if (!userId) {
      return res.status(401).json({ error: 'Not authenticated' });
    }
    
    // Log interaction
    await db.query(`
      INSERT INTO user_interactions (
        user_id, 
        interaction_type, 
        article_title,
        article_id,
        source,
        time_spent_seconds,
        scroll_depth,
        created_at
      ) VALUES ($1, $2, $3, $4, $5, $6, $7, NOW())
    `, [userId, type, articleTitle, articleId, source, timeSpent, scrollDepth]);
    
    // If negative feedback, update user profile
    if (type === 'not_relevant') {
      // This will be used to improve future curation
      console.log(`üìä User ${userId} marked "${articleTitle}" as not relevant`);
      // TODO: Update user profile or model based on this feedback
    }
    
    res.json({ success: true });
    
  } catch (error: any) {
    console.error('Interaction tracking error:', error);
    res.status(500).json({ error: error.message });
  }
});

// Get user's interaction history (for analytics)
router.get('/api/interactions/stats', async (req, res) => {
  try {
    const userId = req.user?.id;
    
    if (!userId) {
      return res.status(401).json({ error: 'Not authenticated' });
    }
    
    const stats = await db.query(`
      SELECT 
        interaction_type,
        COUNT(*) as count,
        AVG(time_spent_seconds) as avg_time
      FROM user_interactions
      WHERE user_id = $1
      AND created_at > NOW() - INTERVAL '30 days'
      GROUP BY interaction_type
    `, [userId]);
    
    res.json(stats.rows);
    
  } catch (error: any) {
    res.status(500).json({ error: error.message });
  }
});

export default router;
```

```sql
CREATE TABLE user_interactions (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id),
  interaction_type VARCHAR(50), -- 'view', 'click', 'save', 'not_relevant', 'share'
  article_id UUID,
  article_title TEXT,
  source VARCHAR(100),
  time_spent_seconds INTEGER,
  scroll_depth DECIMAL,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_interactions_user ON user_interactions(user_id);
CREATE INDEX idx_interactions_type ON user_interactions(interaction_type);
CREATE INDEX idx_interactions_date ON user_interactions(created_at);
```
