import { useRouter } from 'next/router';

  Sparkles, 
  BookOpen, 
  Settings, 
  Workflow,

  ArrowRight,
  Loader2

              <span className="text-white font-bold">K</span>
            </div>
            <span className="text-xl font-semibold text-slate-900">Kurasi</span>
          </div>

          <nav className="space-y-1">
            <a href="/" className="flex items-center gap-3 px-3 py-2.5 rounded-lg bg-slate-200/70 text-slate-900 font-medium text-sm">
              <Sparkles className="w-4 h-4" />
              Mulai
            </a>
            <a href="/how-it-works" className="flex items-center gap-3 px-3 py-2.5 rounded-lg text-slate-500 hover:bg-slate-100 text-sm">
              <Workflow className="w-4 h-4" />
              Cara Kerja
            </a>
            <a href="/dashboard" className="flex items-center gap-3 px-3 py-2.5 rounded-lg text-slate-500 hover:bg-slate-100 text-sm">
              <BookOpen className="w-4 h-4" />
              Brief Saya
            </a>
            <a href="/settings" className="flex items-center gap-3 px-3 py-2.5 rounded-lg text-slate-500 hover:bg-slate-100 text-sm">
              <Settings className="w-4 h-4" />
              Pengaturan

        <div className="p-5 border-t border-slate-200">

            Masuk

              Brief Personal Anda

            <button
              type="submit"

            >
              {loading ? (
                <>
                  <Loader2 className="w-4 h-4 animate-spin" />
                  Menyiapkan...

              ) : (
                <>
                  Buat Brief Saya

                </>

            </button>

          </form>

            Brief pertama akan dikirim ke email Anda besok pagi.

## ADD this API route: /api/onboarding/generate-profile.ts

```typescript
// server/routes/onboarding.ts or pages/api/onboarding/generate-profile.ts

import OpenAI from 'openai';

const deepseek = new OpenAI({
  apiKey: process.env.DEEPSEEK_API_KEY,
  baseURL: 'https://api.deepseek.com/v1'
});

export default async function handler(req: any, res: any) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    const { role, topics, entities, sources } = req.body;

    // Generate personalized system prompt using DeepSeek (cheap & fast)
    const prompt = `Generate a personalized news brief system prompt for this user:

ROLE: ${role}
TOPICS TO MONITOR: ${topics}
PEOPLE/COMPANIES TO TRACK: ${entities || 'Not specified'}
TRUSTED SOURCES: ${sources || 'Not specified'}

Create a JSON response with:
{
  "role": "extracted role",
  "industry": "detected industry",
  "primaryTopics": ["array", "of", "topics"],
  "entitiesToTrack": ["array", "of", "entities"],
  "trustedSources": ["array", "of", "sources"],
  "councilSystemPrompt": "Write a 300-400 word system prompt for 6 AI models searching news for this specific user. Include: who they are, what topics to prioritize, what entities to track, what sources to prefer, and what makes news RELEVANT for them. Be specific and actionable."
}

Respond with valid JSON only.`;

    const result = await deepseek.chat.completions.create({
      model: 'deepseek-chat',
      messages: [
        { role: 'system', content: 'Generate user profile JSON. Respond with valid JSON only, no markdown.' },
        { role: 'user', content: prompt }
      ],
      temperature: 0.3,
      max_tokens: 1500
    });

    const content = result.choices[0].message.content || '{}';
    const profile = JSON.parse(content.replace(/```json\n?|\n?```/g, '').trim());

    res.json({ profile });

  } catch (error: any) {
    console.error('Profile generation error:', error);
    res.status(500).json({ error: error.message });
  }
}
```
